#### 1.3 Composables 设计

**useDownload.ts（下载逻辑）**
```typescript
import { ref } from 'vue';
import axios from 'axios';

export function useDownload() {
  const url = ref('');
  const videoInfo = ref(null);
  const progress = ref(0);
  const status = ref('idle');
  const taskId = ref(null);

  // 获取视频信息
  const getInfo = async () => {
    const response = await axios.post('/api/info', { url: url.value });
    videoInfo.value = response.data;
  };

  // 开始下载
  const startDownload = async (options) => {
    const response = await axios.post('/api/download', {
      url: url.value,
      ...options
    });
    taskId.value = response.data.task_id;
    status.value = 'downloading';

    // 建立 WebSocket 连接
    connectWebSocket(taskId.value);
  };

  // WebSocket 连接
  const connectWebSocket = (id) => {
    const ws = new WebSocket(`ws://localhost:8000/api/ws/${id}`);
    ws.onmessage = (event) => {
      const data = JSON.parse(event.data);
      progress.value = data.progress;
      if (data.status === 'completed') {
        status.value = 'completed';
      }
    };
  };

  return {
    url,
    videoInfo,
    progress,
    status,
    getInfo,
    startDownload
  };
}
```

#### 2.2 服务层设计

**YTDLPService（yt-dlp 封装）**
```python
class YTDLPService:
    """
    yt-dlp 服务类

    职责：
    - 封装 yt-dlp 的复杂配置
    - 提供简洁的 API 接口
    - 处理进度回调
    - 格式化视频信息
    """

    def get_video_info(self, url: str) -> VideoInfo:
        """获取视频信息（不下载）"""
        ydl_opts = {
            'quiet': True,
            'skip_download': True,
        }
        with yt_dlp.YoutubeDL(ydl_opts) as ydl:
            info = ydl.extract_info(url, download=False)
            return self._format_info(info)

    def download_video(self, url, format, quality, progress_callback):
        """下载视频"""
        ydl_opts = {
            'format': self._build_format(format, quality),
            'progress_hooks': [self._build_hook(progress_callback)],
        }
        with yt_dlp.YoutubeDL(ydl_opts) as ydl:
            info = ydl.extract_info(url, download=True)
            return ydl.prepare_filename(info)

    def get_direct_url(self, url, format, quality):
        """获取视频直链"""
        ydl_opts = {'format': self._build_format(format, quality)}
        with yt_dlp.YoutubeDL(ydl_opts) as ydl:
            info = ydl.extract_info(url, download=False)
            return info.get('url', '')
```

**TaskManager（任务管理）**
```python
class TaskManager:
    """
    任务管理器

    职责：
    - 管理下载任务的生命周期
    - 存储任务状态
    - 提供任务查询接口
    - 清理过期任务
    """

    def __init__(self):
        self.tasks: Dict[str, DownloadTask] = {}

    def create_task(self, url: str) -> str:
        """创建新任务"""
        task_id = str(uuid.uuid4())
        task = DownloadTask(task_id=task_id, url=url)
        self.tasks[task_id] = task
        return task_id

    def get_task(self, task_id: str) -> DownloadTask:
        """获取任务"""
        return self.tasks.get(task_id)

    def update_task(self, task_id: str, **kwargs):
        """更新任务状态"""
        task = self.tasks[task_id]
        for key, value in kwargs.items():
            setattr(task, key, value)
```

## 性能优化

### 前端优化
1. **代码分割**：使用动态导入减少初始加载
2. **懒加载**：视频信息组件按需加载
3. **缓存策略**：使用 localStorage 缓存视频信息

### 后端优化
1. **异步处理**：使用 FastAPI 的异步特性
2. **后台任务**：下载操作在后台执行，不阻塞请求
3. **内存清理**：定期清理已完成任务

### 网络优化
1. **直链下载**：前端直接从视频平台 CDN 下载
2. **SSE 流式输出**：AI 问答按 token 增量返回，降低首字等待时间
3. **压缩传输**：启用 gzip 压缩
